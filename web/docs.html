# Speed Control - Modify all channels
$ curl -X POST http://{{ .Configuration.ListenAddress }}:{{ .Configuration.ListenPort }}/api/speed -d '{ "deviceId":"5C126A3EB51A39569ABADC4C3A1FCF54", "channelId":0, "profile":"Performance" }' --silent | jq
{
  "code": 200,
  "status": 1,
  "message": "Device speed successfully changed"
}
